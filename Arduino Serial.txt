// sketch_LGT8F328P_Character_Map_2_TESTA
// include to be able change the prescaler frequency 32Mhz
#include <avr/power.h>

void setup() {
    clock_prescale_set(clock_div_1);
    Serial.begin(9600);
    while (!Serial) {
      ;  // wait for serial port to connect. Needed for native USB port only
  }
}
int mASCII = 33;             // start of ASCII character
int mLineCount = 1;          // line counter

void loop() {

if (mASCII != 127) {
    char myChar = mASCII;
    Serial.print(myChar);
    mASCII++;
if (mASCII == 127) {          //
    Serial.println();         // must be here to work
  }                           //
 } 

else if (mLineCount != 5) {
    mLineCount++;
    mASCII = 33;
 }
}

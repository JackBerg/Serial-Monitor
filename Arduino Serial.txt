// sketch_LGT8F328P_Character_Map_2
// include to be able change the prescaler frequency 32Mhz
// replace board.txt
#include <avr/power.h>       // to enable 32mHz

void setup() {
  clock_prescale_set(clock_div_1); // to enable 32mHz
  Serial.begin(9600);
  while (!Serial) {
      ;  // wait for serial port to connect. Needed for native USB port only
  }
}
int mASCII = 33;             // 33 = first ASCII character "!",126 = last ASCII character "~"
int mLineCount = 1;          // line counter

void loop() {

if (mASCII != 127) {         // send ASCII under 127
    char myChar = mASCII;    // convert mASCII to char 
    Serial.print(myChar);    // print mASCII 
    mASCII++;                // increase mASCII 
 } 

else if (mLineCount != 5) {  // true if line mLineCount not = 5
    mLineCount++;            // increase mLineCount
if (mASCII == 127) {         // true if ASCII = 127
    mASCII = 33;             // reset ASCII line 
    Serial.println();        // add CRLF
   }   
  }
}

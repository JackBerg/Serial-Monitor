//sketch_LGT8F328P_Character_Map_2
// include to be able change the prescaler frequency 32Mhz
#include <avr/power.h>

void setup() {
  clock_prescale_set(clock_div_1);
  Serial.begin(9600);
while (!Serial) {
    ;  // wait for serial port to connect. Needed for native USB port only
  }
}
int mASCII = 33;              // start of ASCII character
int mLineCount = 1;           // line counter

void loop() {
   if (mASCII != 127) {       // send ASCII under 127
    Serial.print(Char(mASCII));
    mASCII++;
if (mASCII == 127) {       
    Serial.println();         // add LF
  }   
 } 
else if (mLineCount != 5) {   // line counter
   mASCII = 33;               // reset ASCII line 
   mLineCount++;
 }
}
